\section{Additional derivations}\label{app:mathematics}


\subsection{Extremal coefficient and exponent measure}\label{app:exponent}

Let \((Z_i,Z_j)\) be bivariate max-stable with unit-Fr\'echet margins and exponent measure \(V_{ij}\),
\[
    \Pr(Z_i\le z_i,Z_j\le z_j)=\exp\{-V_{ij}(z_i,z_j)\},
\]
where \(V_{ij}\) is homogeneous of order \(-1\), \(V_{ij}(az_i,az_j)=a^{-1}V_{ij}(z_i,z_j)\). The extremal coefficient \(\theta_{ij}\) is defined on the diagonal by
\[
    \Pr(Z_i\le z,Z_j\le z)=\Pr(Z_i\le z)^{\theta_{ij}}=\exp(-\theta_{ij}/z).
\]
Homogeneity with \(a=z\) gives \(V_{ij}(z,z)=z^{-1}V_{ij}(1,1)\), so the diagonal probability also equals \(\exp\{-V_{ij}(1,1)/z\}\). Matching the two exponents,
\[
    \theta_{ij}=V_{ij}(1,1).
\]
The extremes bound it: under complete dependence \(Z_i=Z_j\), the diagonal probability is \(\exp(-1/z)\), giving \(\theta_{ij}=1\); under independence it is \(\exp(-2/z)\), giving \(\theta_{ij}=2\). Hence \(1\le\theta_{ij}\le2\), with smaller values meaning stronger extremal dependence.

\subsection{Brown--Resnick extremal coefficient}\label{app:br-theta}

Under the Brown--Resnick model the pairwise exponent measure is of H\"usler--Reiss form,
\[
    V_d(z_i,z_j)=\frac{1}{z_i}\,\Phi\!\Bigl(\frac{a_c(d)}{2}+\frac{1}{a_c(d)}\log\frac{z_j}{z_i}\Bigr)
    +\frac{1}{z_j}\,\Phi\!\Bigl(\frac{a_c(d)}{2}+\frac{1}{a_c(d)}\log\frac{z_i}{z_j}\Bigr),
\]
with \(a_c(d)=\sqrt{\gamma_c(d)}=(d/\rho_c)^{\alpha_c/2}\) for the powered-exponential variogram \(\gamma_c(d)=(d/\rho_c)^{\alpha_c}\) (Section~\ref{sec:meth-variogram}). Evaluating at \(z_i=z_j=1\), the logarithms vanish and both normal terms reduce to \(\Phi(a_c(d)/2)\); using \(\theta_{ij}=V_{ij}(1,1)\) from Appendix~\ref{app:exponent},
\[
    \theta_c(d)=V_d(1,1)=2\,\Phi\!\Bigl(\frac{a_c(d)}{2}\Bigr)=2\,\Phi\!\Bigl(\tfrac12 (d/\rho_c)^{\alpha_c/2}\Bigr).
\]
This curve increases in \(d\): as \(d\to0\), \(a_c(d)\to0\) and \(\theta_c(d)\to2\Phi(0)=1\); as \(d\to\infty\), \(a_c(d)\to\infty\) and \(\theta_c(d)\to2\). At \(d=\rho_c\) the variogram equals one, so \(a_c(\rho_c)=1\) and \(\theta_c(\rho_c)=2\Phi(1/2)\approx1.38\); this fixes the reading of \(\rho_c\) as a distance scale used in Section~\ref{sec:meth-comparison}.

\subsection{Finite-level tail dependence}\label{app:tail-dependence}

On the uniform scale a max-stable pair satisfies \(\Pr(U_i\le u,U_j\le u)=u^{\theta_{ij}}\), obtained from Appendix~\ref{app:exponent} by evaluating the diagonal at \(z=-1/\log u\). The finite-level tail-dependence coefficient used as a diagnostic (Section~\ref{sec:meth-diagnostics}) is
\[
    \chi_{u,ij}=\Pr(U_j>u\mid U_i>u)=\frac{\Pr(U_i>u,U_j>u)}{\Pr(U_i>u)}.
\]
With \(\Pr(U_i>u)=1-u\) and, by inclusion--exclusion, \(\Pr(U_i>u,U_j>u)=1-2u+u^{\theta_{ij}}\),
\[
    \chi_{u,ij}=\frac{1-2u+u^{\theta_{ij}}}{1-u}.
\]
As \(u\uparrow1\) both numerator and denominator vanish, and l'H\^opital's rule gives
\[
    \chi_{ij}=\lim_{u\uparrow1}\frac{-2+\theta_{ij}\,u^{\theta_{ij}-1}}{-1}=2-\theta_{ij},
\]
so \(\theta_{ij}=1\) is asymptotic dependence (\(\chi_{ij}=1\)) and \(\theta_{ij}=2\) asymptotic independence (\(\chi_{ij}=0\)). Substituting the Brown--Resnick curve gives the model-implied finite-level coefficient \(\chi^{\mathrm{BR}}_{u}(d)=\{1-2u+u^{\theta_c(d)}\}/(1-u)\), which Section~\ref{sec:meth-diagnostics} compares with the empirical \(\widehat\chi_{u,ij}^{(c)}\) at \(u=0.95\).
